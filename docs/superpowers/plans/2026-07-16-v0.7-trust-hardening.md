**Goal:** Harden RC Central's trust surface for v0.7 — hash-verified self-updates, pinned installer hashes, a catalog traversal guard and shape check — plus the overdue README refresh, CI action bumps, and a ruff lint gate.

### Task 3: Traversal guard in `installer._find_exe`

`_find_exe` currently returns `root / relative` unchecked (`app/installer.py:312`), so a hostile catalog hint like `..\..\evil.exe` escapes the tool directory. Archive extraction is already sanitized; this is the one unguarded path.

**Interfaces:**
- Consumes: existing `installer._find_exe(root: Path, relative: str | None) -> Path`, `installer.ExeNotFound`.
- Produces: same signature; the returned path is now resolved (absolute).

- [ ] **Step 2: Run it to verify it fails**

- [ ] **Step 3: Add the guard**

In `app/installer.py`, change the head of `_find_exe` from:

```python
def _find_exe(root: Path, relative: str | None) -> Path:
    """Resolve the tool's exe: catalog hint first, else the single plausible exe."""

        return root / relative
```

to:

```python
def _find_exe(root: Path, relative: str | None) -> Path:
    """Resolve the tool's exe: catalog hint first, else the single plausible exe."""
    if relative:
        candidate = (root / relative).resolve()
        # the hint comes from the remote catalog: never let it escape the tool dir
        if candidate.is_relative_to(root.resolve()) and candidate.exists():
            return candidate
```

The candidate-scan fallback below stays unchanged.

### Task 4: Shape-check the fetched catalog before trusting or caching it

`catalog.load_catalog` caches and returns whatever the remote sends as long as it parses as JSON (`app/catalog.py:22`). A wrong-shaped payload (or a corrupt cache file from disk trouble) should be treated like a failed fetch: fall through to cache/bundled, never poison the cache.

**Files:**
- Modify: `app/catalog.py` (`load_catalog` + new `_valid` helper)
- Test: `tests/test_core.py` (append at end of file)

**Interfaces:**
- Consumes: existing `catalog.load_catalog() -> list[dict]`, module attrs `CATALOG_URL`, `DATA_DIR`, `CACHE_FILE`.
- Produces: `catalog._valid(tools) -> bool` (module-private; tests may reference it).

- [ ] **Step 1: Write the failing tests**

Append to `tests/test_core.py`:

```python

    class _Resp:
        def raise_for_status(self):
            pass

        def json(self):
            return {"not": "a tool list"}  # valid JSON, wrong shape

    monkeypatch.setattr(catalog.requests, "get", lambda *a, **k: _Resp())
    monkeypatch.setattr(catalog, "DATA_DIR", tmp_path)
    monkeypatch.setattr(catalog, "CACHE_FILE", tmp_path / "catalog.json")

    tools = catalog.load_catalog()

    # fell through to the bundled catalog, and did not cache the bad payload
    assert isinstance(tools, list) and tools
    assert all("id" in t for t in tools)
    assert not (tmp_path / "catalog.json").exists()

def _patch_sha_flow(monkeypatch, tmp_path, payload, sha_line):
    api = FakeResp(json_data={"tag_name": "v99.0.0", "assets": _sha_assets(payload)})
    responses = {
        "https://dl/exe": FakeResp(content=payload),
        "https://dl/sha": FakeResp(content=sha_line),

        updater.requests, "get", lambda url, **kw: responses.get(url, api)

    monkeypatch.setattr(updater, "DATA_DIR", tmp_path)
    monkeypatch.setattr(updater, "PENDING", pending)
    monkeypatch.setattr(updater, "_staged_version", None)
    return pending


def test_fetch_update_rejects_sha256_mismatch(monkeypatch, tmp_path):
    payload = b"MZ" + b"x" * 64
    wrong = b"0" * 64 + b"  RCCentral-windows-x64.exe\n"

    assert not pending.exists()
    assert list(tmp_path.glob("*.part")) == []


def test_fetch_update_accepts_matching_sha256(monkeypatch, tmp_path):
    import hashlib

    payload = b"MZ" + b"x" * 64
    digest = hashlib.sha256(payload).hexdigest().encode()
    good = digest + b"  RCCentral-windows-x64.exe\n"
    pending = _patch_sha_flow(monkeypatch, tmp_path, payload, good)

    assert updater.fetch_update(force=True) is True

def _release_sha256(assets: list[dict], asset_name: str) -> str | None:
    """Published hex digest for an asset (its '<name>.sha256' release file), if any."""
    meta = next((a for a in assets if a.get("name") == asset_name + ".sha256"), None)
    if meta is None:
        return None
    resp = requests.get(meta["browser_download_url"], timeout=10)
    resp.raise_for_status()
    parts = resp.text.split()  # "<hex>  <filename>" (sha256sum format)
    return parts[0].lower() if parts else None
```
